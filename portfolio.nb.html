library(tidyverse)
library(dplyr)
library(ggplot2)
data4 <- readxl::read_xls("aav8130_Data-S4.xls")
d4 <- data4 %>% select(`Cell type`, `Gene name`, `Fold change`, `SFARI gene`) 
neuronal = c("L2/3","L4","L5/6","L5/6-CC","Neu-mat","Neu-NRGN-I","Neu-NRGN-II","IN-PV","IN-SST","IN-VIP","IN-SV2C" )
d_4 = d4 %>% mutate('neuronal'= `Cell type` %in% neuronal) %>% 
  mutate(`SFARI gene`=ifelse(`SFARI gene`=='yes','SFARI','Non-SFARI'),
         neuronal=ifelse(neuronal=='TRUE','Neuro','Non-Neuro')) %>%
  mutate(neuronal_SFARI=paste(`SFARI gene`,neuronal,sep = '/'))
p <-d_4 %>% ggplot(aes(`Gene name`,`Fold change`,col=`neuronal_SFARI`)) +        geom_point(size=0.5) +
  geom_hline(yintercept = 0.0, color= '#D7D6C6', alpha=0.7)+
  facet_wrap(~`Cell type`) + 
  theme(axis.text.x = element_blank(),
        axis.ticks = element_blank(),
        axis.text.y.left = element_text(face='bold',size=5))
